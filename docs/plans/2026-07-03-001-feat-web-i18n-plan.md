# Web i18n (native @angular/localize, es-first) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make `@hsm/web` render in Spanish by default with a runtime (reload-based) switch to English, using Angular's first-party i18n, and localize API-surfaced messages via stable backend codes.

**Architecture:** Native `@angular/localize` with **Spanish (`es-EC`) as the source locale** authored directly in templates; **English (`en`)** as a translation target. `ng build --localize` emits one bundle per locale under a base-href (`/es/`, `/en/`); a `LanguageService` switches by reloading into the other locale's build. Backend attaches a stable `ApiErrorCode` to every error envelope; the frontend maps codes → `$localize` messages.

**Tech Stack:** Angular 21 (standalone, zoneless, `@angular/build:application` builder), PrimeNG 21, `@angular/localize`, NestJS backend, `@hsm/common` shared package, pnpm monorepo, `serve` static server.

## Global Constraints

- Source locale is `es-EC`; first target is `en`. `pt` is **not** built now (structure must not preclude it). — spec R1, Non-Goals
- Every user-facing string in `apps/frontend/web/src` carries a **stable custom ID** of the form `@@<domain>.<feature>.<key>` and is authored in **Spanish**. — spec R2, KTD3
- Default served locale is `es`; a user's explicit choice persists in `localStorage` under key `hsm.lang`. — spec R1, R3, R4
- Supported locales are exactly `['es', 'en']`; an unknown/absent value resolves to `es`. — spec R3, R4
- Default currency is **USD**; dates/numbers follow the active locale. — spec R6
- Backend error envelope always includes a stable `issue.code`; validation errors also include `issue.errors: { field, constraints: string[] }` (constraint **keys**). — spec R9, KTD8
- Frontend never displays a raw backend English string or a blank for a known error surface: an unknown code falls back to a generic localized message. — spec R10
- Existing frontend specs (247 baseline) and backend specs (380 baseline) must stay green.
- Angular project name is `web`; source root `apps/frontend/web/src`; build builder `@angular/build:application`.

import localeEsEc from '@angular/common/locales/es-EC';

export type AppLocale = 'es' | 'en';

/**
 * The locale baked into THIS build. In a localized build `$localize.locale` is

 * our two-letter app locale.

export function activeLocale(): AppLocale {

}

export function registerAppLocales(): void {

}

Create `apps/frontend/web/src/app/core/i18n/language.service.ts`:
```ts
import { Injectable, signal } from '@angular/core';
import { type AppLocale, activeLocale } from './locale-init';

export const LANG_STORAGE_KEY = 'hsm.lang';
const SUPPORTED = ['es', 'en'] as const;

function isSupported(v: string | null | undefined): v is AppLocale {
  return v === 'es' || v === 'en';
}

/** Pure resolver used pre-bootstrap (no Angular DI). URL prefix wins, else storage, else es. */
export function resolveBootLocale(
  pathname: string,
  stored: string | null,
): AppLocale {
  const seg = pathname.split('/').filter(Boolean)[0];
  if (isSupported(seg)) return seg;
  return isSupported(stored) ? stored : 'es';
}

@Injectable({ providedIn: 'root' })
export class LanguageService {
  readonly SUPPORTED = SUPPORTED;
  private readonly _current = signal<AppLocale>(activeLocale());
  readonly current = this._current.asReadonly();

  /** Persist choice and reload into that locale's build, preserving the route. */
  switch(locale: AppLocale): void {
    if (!isSupported(locale)) return;
    localStorage.setItem(LANG_STORAGE_KEY, locale);
    const path = window.location.pathname.replace(/^\/(es|en)(?=\/|$)/, '');
    const target = `/${locale}${path || '/'}${window.location.search}`;
    window.location.assign(target);
  }
}
```

import { resolveBootLocale, LANG_STORAGE_KEY } from './app/core/i18n/language.service';

// If the URL has no locale prefix, redirect into the persisted/default locale
// build. The static server serves each locale under /es/ and /en/.
{
  const first = window.location.pathname.split('/').filter(Boolean)[0];
  if (first !== 'es' && first !== 'en') {
    const locale = resolveBootLocale(
      window.location.pathname,
      localStorage.getItem(LANG_STORAGE_KEY),
    );
    window.location.replace(
      `/${locale}${window.location.pathname}${window.location.search}`,
    );
  }
}
```

import { ApiErrorCode } from '@hsm/common/enums';

export function apiMessage(code: string): string {
  switch (code) {
    case ApiErrorCode.InvalidCredentials:
      return $localize`:@@api.error.invalidCredentials:Usuario o contraseña incorrectos`;
    case ApiErrorCode.Unauthorized:
      return $localize`:@@api.error.unauthorized:Sesión no autorizada`;
    case ApiErrorCode.Forbidden:
      return $localize`:@@api.error.forbidden:No tienes permiso para esta acción`;
    case ApiErrorCode.NotFound:
      return $localize`:@@api.error.notFound:Recurso no encontrado`;
    case ApiErrorCode.Validation:
      return $localize`:@@api.error.validation:Revisa los datos ingresados`;
    default:
      return $localize`:@@api.error.unexpected:Ocurrió un error inesperado`;
  }
}

export function validationMessage(constraintKey: string): string {
  switch (constraintKey) {
    case 'isNotEmpty':
      return $localize`:@@api.validation.required:Campo obligatorio`;
    case 'isEmail':
      return $localize`:@@api.validation.email:Correo inválido`;
    case 'minLength':
      return $localize`:@@api.validation.minLength:Demasiado corto`;
    case 'maxLength':
      return $localize`:@@api.validation.maxLength:Demasiado largo`;
    default:
      return $localize`:@@api.validation.invalid:Valor inválido`;
  }
}
```
